from flask import Flask, render_template, jsonify
from flask_cors import CORS
import psutil
import subprocess

app = Flask(__name__)
CORS(app)

stress_process = None

@app.route('/')
def index():
    return render_template('index.html')

@app.route('/cpu_percentage')
def cpu_percentage():
    return jsonify(cpu=psutil.cpu_percent(interval=1))

@app.route('/increase_load')
def increase_load():
    global stress_process

    if not stress_process:
        # Strong CPU Load
        stress_process = subprocess.Popen([
            'stress',
            '--cpu', '8',
            '--timeout', '600'
        ])

    return jsonify(status='CPU Load Increased')

@app.route('/cancel_load')
def cancel_load():
    global stress_process

    subprocess.run(['pkill', 'stress'])
    stress_process = None

    return jsonify(status='CPU Load Cancelled')

if __name__ == "__main__":
    app.run(host='0.0.0.0', port=80)
EOF
